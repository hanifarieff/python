162def combine_mpi_ehr(source,employee_admission,occupation_patient):
163    """ Join antara data dari MPI dan EHR, lalu buat flag apakah Pasien merupakan pegawai RSCM atau tidak"""
164    source = source.merge(employee_admission,how='left',on='PatientID').merge(occupation_patient,how='left',on='jobclass_id')
165
166    new_order_columns = ['PatientID','PersonID','Confidentiality','PatientStatus','MigrationID','MedicalNo','PatientName','BirthDate',
167                        'Gender','MaritalStatus','ReligionID','NIK','Address','PatientCreatedDate','PatientUpdatedDate',
168                        'PatientNullifiedDate','RegionalCode','PersonStatusCode','PatientStatusCode','PersonTitle','FamilyName','ExternalID',
169                        'PassportNo','FatherName','MotherName','SpouseName','PlaceOfBirth','PostalCode','Race','Education','Occupation',
170                        'BloodType','Nationality','TelephoneNo','PhoneNo','Email','EmployeeID','IsEmployee']
171    source = source.reindex(columns=new_order_columns)
172
173    #bikin kondisi untuk kolom IsEmployee kalo EmployeeID null = 0, tidak null = 1
174    source.loc[source['EmployeeID'].isnull(),'IsEmployee']=0
175    source.loc[source['EmployeeID'].notnull(),'IsEmployee']=1
176
177    # rubah tipe data EmployeeID dan IsEmployee jadi integer
178    source['EmployeeID'] = source['EmployeeID'].fillna(0).astype('int64')
179    source['IsEmployee'] = source['IsEmployee'].fillna(0).astype('int64')
180    source['Occupation'].fillna('-',inplace=True)
181    return source

237def detect_changes(source, target):
238    """ Deteksi perubahan antara dataframe `source` dan `target` """
239    
240    # ambil dataframe yang mengalami perubahan (termasuk data update dan data new)
241    change = source[~source.apply(tuple,1).isin(target.apply(tuple,1))]
242
243    # ambil data dari variable change ambil primary key PatientID cek dengan variable target, masukkan ke variable modified
244    modified = change[change[['PatientID']].apply(tuple,1).isin(target[['PatientID']].apply(tuple,1))]
245    
246    # tambahkan kolom ScdStart isi dengan tanggal, dan kolom ScdActive beri nilai 1
247    modified['ScdStart']=date
248    modified['ScdActive']=1
249
250    # ambil data dari variable change, ambil primary key PatientID dan variable target, masukkan ke variable inserted, kebalikan dari update pake simbol ~
251    inserted =  change[~change[['PatientID']].apply(tuple,1).isin(target[['PatientID']].apply(tuple,1))]
252    inserted['ScdStart']=date
253    inserted['ScdActive']=1
254
255    return modified, inserted
